

Installation:

- alle alfred-Dateien und ogm-monitor.service nach /etc/systemd/system kopieren
- Ordner /etc/dnsmasq.d nach /etc/dnsmasq.d kopieren
  (mesh-br0.conf -> die IP hinter "dhcp-option=3," muss der Node IP entsprechen!)
- sudo apt install alfred
- sudo alfred -i bat0 -b

- sudo apt install dnsmasq
- echo 'net.ipv4.ip_forward=1' | sudo tee /etc/sysctl.d/99-ipforward.conf
- sudo sysctl -p /etc/sysctl.d/99-ipforward.conf
- echo 'natak ALL=(root) NOPASSWD:/bin/systemctl restart dnsmasq, \
  /bin/systemctl restart reticulum, \
  /bin/systemctl restart systemd-networkd' | sudo tee /etc/sudoers.d/mesh-monitor
- sudo chmod 440 /etc/sudoers.d/mesh-monitor
-sudo tail -f /var/lib/misc/dnsmasq.leases

- sudo systemctl daemon-reload
- sudo systemctl enable --now dnsmasq
- sudo systemctl enable --now alfred.service
- sudo systemctl enable --now alfred-hostname.timer
- sudo systemctl enable --now ogm-monitor.service
